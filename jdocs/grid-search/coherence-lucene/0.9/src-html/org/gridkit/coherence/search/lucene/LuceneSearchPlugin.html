017    package org.gridkit.coherence.search.lucene;
018    
019    import org.apache.lucene.analysis.Analyzer;
020    import org.apache.lucene.search.Query;
021    import org.apache.lucene.store.Directory;
022    import org.gridkit.coherence.search.IndexEngineConfig;
023    import org.gridkit.coherence.search.IndexInvocationContext;
024    import org.gridkit.coherence.search.IndexUpdateEvent;
025    import org.gridkit.coherence.search.PlugableSearchIndex;
026    
027    import java.io.Serializable;
028    import java.util.Map;
029    import java.util.Set;
030    
031    /**
032     * @author Alexey Ragozin (alexey.ragozin@gmail.com)
033     */
034    public class LuceneSearchPlugin implements PlugableSearchIndex<LuceneInMemoryIndex, LuceneIndexConfig, Query>, Serializable {
035    
036            private String name = "";
037            
038            public LuceneSearchPlugin() {
039            }
040            
041            public LuceneSearchPlugin(String name) {
042                    this.name = name;
043            }
044    
045            @Override
046            public Object createIndexCompatibilityToken(LuceneIndexConfig indexConfig) {
047                    return new GenericNamedLuceneIndexToken(name);
048            }
049    
050            @Override
051            public void configure(IndexEngineConfig config) {
052                    config.setOldValueOnUpdateEnabled(false);
053            }
054    
055            @Override
056            public LuceneInMemoryIndex createIndexInstance(LuceneIndexConfig indexConfig) {
057                    Directory directory = indexConfig.createDirectoryInstance();
058                    Analyzer analyzer = indexConfig.getAnalyzer(); 
059                    return new LuceneInMemoryIndex(directory, analyzer);
060            }
061    
062            @Override
063            public void updateIndexEntries(LuceneInMemoryIndex index, Map<Object, IndexUpdateEvent> events, IndexInvocationContext context) {
064                    index.update(events, context);
065            }
066    
067            @Override
068            public boolean applyIndex(LuceneInMemoryIndex index, Query query, final Set<Object> keySet, IndexInvocationContext context) {
069                    index.applyIndex(query, keySet, context);
070                    return false;
071            }
072    
073            @Override
074            public int calculateEffectiveness(LuceneInMemoryIndex index, Query query, Set<Object> keySet, IndexInvocationContext context) {
075                    // lucene index does not support evaluate method, so index should be used in all cases
076                    return 1;
077            }
078    
079            @Override
080            public boolean evaluate(Query query, Object document) {
081                    // TODO we can create new new index put single document and query this index
082                    return false;
083            }
084    }
